# SQL a agregar: seguimiento de planificacion por alumno

Este cambio agrega **2 tablas nuevas** para que el alumno pueda:
- elegir semana y dia,
- ver ejercicios planificados,
- cargar por ejercicio: peso, repeticiones y RPE,
- mantener historial sin romper el esquema actual.

## SQL

```sql
-- =========================================
-- 1) Sesiones de entrenamiento del alumno
-- =========================================
create table if not exists public.entrenamiento_sesiones (
  id bigint generated by default as identity primary key,
  alumno_id bigint not null references public.alumnos(id) on delete cascade,
  planificacion_id bigint not null references public.planificaciones(id) on delete cascade,
  hoja_id bigint not null references public.planificacion_hojas(id) on delete restrict,
  dia_id bigint not null references public.planificacion_dias(id) on delete restrict,
  semana smallint not null check (semana between 1 and 52),

  estado text not null default 'abierta'
    check (estado in ('abierta', 'completada')),

  fecha_entrenamiento date null,
  iniciado_at timestamptz not null default now(),
  finalizado_at timestamptz null,

  created_at timestamptz not null default now()
);

create unique index if not exists ux_entrenamiento_sesiones_scope
  on public.entrenamiento_sesiones (alumno_id, planificacion_id, hoja_id, dia_id, semana);

create index if not exists ix_entrenamiento_sesiones_alumno_recent
  on public.entrenamiento_sesiones (alumno_id, created_at desc);

create index if not exists ix_entrenamiento_sesiones_lookup
  on public.entrenamiento_sesiones (planificacion_id, hoja_id, dia_id, semana);

-- =========================================
-- 2) Registros por ejercicio en la sesion
-- =========================================
create table if not exists public.entrenamiento_registros (
  id bigint generated by default as identity primary key,
  sesion_id bigint not null references public.entrenamiento_sesiones(id) on delete cascade,
  planificacion_ejercicio_id bigint not null references public.planificacion_ejercicios(id) on delete restrict,
  ejercicio_id bigint not null references public.ejercicios(id) on delete restrict,

  peso_kg numeric(6,2) null check (peso_kg is null or (peso_kg >= 0 and peso_kg < 1000)),
  repeticiones smallint null check (repeticiones is null or (repeticiones >= 0 and repeticiones <= 500)),
  rpe numeric(3,1) null check (rpe is null or (rpe >= 0 and rpe <= 10)),
  notas text null,

  series jsonb not null default '[]'::jsonb,

  ejercicio_nombre_snapshot text not null,
  categoria_snapshot text null,
  prescripcion_dosis text null,
  prescripcion_rpe smallint null check (prescripcion_rpe is null or (prescripcion_rpe >= 0 and prescripcion_rpe <= 10)),

  created_at timestamptz not null default now()
);

create unique index if not exists ux_entrenamiento_registros_sesion_plan_ej
  on public.entrenamiento_registros (sesion_id, planificacion_ejercicio_id);

create index if not exists ix_entrenamiento_registros_sesion
  on public.entrenamiento_registros (sesion_id);

create index if not exists ix_entrenamiento_registros_plan_ej
  on public.entrenamiento_registros (planificacion_ejercicio_id);

-- =========================================
-- 3) Indices recomendados en tablas actuales
-- =========================================
create index if not exists ix_planificacion_dias_hoja_orden
  on public.planificacion_dias (hoja_id, orden);

create index if not exists ix_planificacion_ejercicios_dia_orden
  on public.planificacion_ejercicios (planificacion_dia_id, orden);

create index if not exists ix_planificacion_semanas_pe_semana
  on public.planificacion_semanas (planificacion_ejercicio_id, semana);
```

## Resultado esperado

- Se agregan **2 tablas** nuevas:
  - `entrenamiento_sesiones`
  - `entrenamiento_registros`
- No se modifica ni rompe el modelo existente de planificacion.
- Queda preparado para evolucionar a series por ejercicio en el futuro (sin migracion destructiva).
